rules_version = '2';

service firebase.storage {
  match /b/{bucket}/o {
    match /memory-battle/{roomCode}/{gameId}/{fileName} {
      allow read: if true;
      allow write: if request.resource == null
        || (request.resource.size < 2 * 1024 * 1024
          && request.resource.contentType.matches('image/(jpeg|png|webp)'));
    }
  }
}
